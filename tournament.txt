        return 'c'

        if their_history[-1]=='b':
            return 'b'
        else:
            return 'c'

        if my_history[-1]=='c' and their_history[-1]=='b':
            return 'b'
        elif my_history[-1]=='b' and their_history[-1]=='b':
            return 'b'
        else:
            return 'c'
    elif len(my_history) > 5:
        last6rounds = str(their_history[-6:])

        for thing in last6rounds:
            letter = thing
            if letter == 'c':
                n=n+1
        
        if n > 3:
            return 'c' # Betray if severely punished last time,
        else:
            return 'b' # otherwise collude--------------------------------------------------------------------------------
